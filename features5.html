// Quickstart: Authorise with PaymentHood
const client = new PaymentHoodSDK({
    apiKey: process.env.PAYMENTHOOD_KEY,
    environment: 'sandbox'
});

const intent = await client.payments.create({
    amount: 12900,
    currency: 'USD',
    captureStrategy: 'delayed',
    customer: { id: 'cust_4391' },
    routingProfile: 'north-america',
    risk: { frictionless3DS: true }
});

console.log('Authorised in', intent.latency, 'ms');
